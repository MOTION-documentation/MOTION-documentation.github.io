507 template <typename UnsignedIntegralType,
508  typename = std::enable_if_t<std::is_unsigned_v<UnsignedIntegralType>>,
509  typename Allocator = std::allocator<std::byte>>
510 UnsignedIntegralType ToOutput(std::vector<BitVector<Allocator>> bit_vectors) {
511  static_assert(std::is_integral<UnsignedIntegralType>::value);
512  static_assert(sizeof(UnsignedIntegralType) <= 8);
513  if constexpr (sizeof(UnsignedIntegralType) == 1) {
514  static_assert(std::is_same_v<UnsignedIntegralType, std::uint8_t>);
515  } else if constexpr (sizeof(UnsignedIntegralType) == 2) {
516  static_assert(std::is_same_v<UnsignedIntegralType, std::uint16_t>);
517  } else if constexpr (sizeof(UnsignedIntegralType) == 4) {
518  static_assert(std::is_same_v<UnsignedIntegralType, std::uint32_t>);
519  } else if constexpr (sizeof(UnsignedIntegralType) == 8) {
520  static_assert(std::is_same_v<UnsignedIntegralType, std::uint64_t>);
521  }
522 
523  // kBitLength is always equal to bit_vectors
524  constexpr auto kBitLength{sizeof(UnsignedIntegralType) * 8};
525 
526  assert(!bit_vectors.empty());
527  if (kBitLength != bit_vectors.size()) {
528  throw std::runtime_error(
529  fmt::format("Trying to convert to different bitlength: is {}, expected {}",
530  bit_vectors.size(), kBitLength));
531  }
532 
533  const auto number_of_simd{bit_vectors.at(0).GetSize()};
534  assert(number_of_simd > 0u);
535  for ([[maybe_unused]] auto i = 0ull; i < bit_vectors.size(); ++i)
536  assert(bit_vectors.at(i).GetSize() == number_of_simd);
537 
538  UnsignedIntegralType output_value{0};
539  // Converting values in a BitVector to UnsignedIntegralType
540  for (auto i = 0ull; i < kBitLength; ++i) {
541  assert(bit_vectors.at(i).GetSize() == 1);
542  output_value += static_cast<UnsignedIntegralType>(bit_vectors.at(i)[0]) << i;
543  }
544 
545  return output_value;
546 }

561 template <typename UnsignedIntegralType,
562  typename = std::enable_if_t<std::is_unsigned_v<UnsignedIntegralType>>,
563  typename Allocator = std::allocator<std::byte>>
564 std::vector<UnsignedIntegralType> ToVectorOutput(std::vector<BitVector<Allocator>> bit_vectors) {
565  static_assert(std::is_integral<UnsignedIntegralType>::value);
566  static_assert(sizeof(UnsignedIntegralType) <= 8);
567  if constexpr (sizeof(UnsignedIntegralType) == 1) {
568  static_assert(std::is_same_v<UnsignedIntegralType, std::uint8_t>);
569  } else if constexpr (sizeof(UnsignedIntegralType) == 2) {
570  static_assert(std::is_same_v<UnsignedIntegralType, std::uint16_t>);
571  } else if constexpr (sizeof(UnsignedIntegralType) == 4) {
572  static_assert(std::is_same_v<UnsignedIntegralType, std::uint32_t>);
573  } else if constexpr (sizeof(UnsignedIntegralType) == 8) {
574  static_assert(std::is_same_v<UnsignedIntegralType, std::uint64_t>);
575  }
576 
577  constexpr auto kBitLength{sizeof(UnsignedIntegralType) * 8};
578 
579  assert(!bit_vectors.empty());
580  if (kBitLength != bit_vectors.size()) {
581  throw std::runtime_error(
582  fmt::format("Trying to convert to different bitlength: is {}, expected {}",
583  bit_vectors.size(), kBitLength));
584  }
585 
586  const auto number_of_simd{bit_vectors.at(0).GetSize()};
587  assert(number_of_simd > 0u);
588  for ([[maybe_unused]] auto i = 0ull; i < bit_vectors.size(); ++i)
589  assert(bit_vectors.at(i).GetSize() == number_of_simd);
590 
591  // Converting values in a BitVector to a value of UnsignedIntegralType
592  // Example: If the jth BitVector of size 3 is represented by the tuple: (xj, yj, zj)
593  // Then output_vector[0] == x0*2^0 + x1*2^1 + ... + xn*2^n
594  // output_vector[1] == y0*2^0 + y1*2^1 + ... + yn*2^n
595  // output_vector[2] == z0*2^0 + z1*2^1 + ... + zn*2^n
596  std::vector<UnsignedIntegralType> output_vector;
597  for (auto i = 0ull; i < number_of_simd; ++i) {
598  UnsignedIntegralType value{0};
599  for (auto j = 0ull; j < kBitLength; ++j) {
600  value += static_cast<UnsignedIntegralType>(bit_vectors.at(j)[i]) << j;
601  }
602  output_vector.emplace_back(value);
603  }
604  return output_vector;
605 }
